UPDATE Courses SET NUM_OF_STUDENTS = (SELECT COUNT(*) FROM Grades WHERE Grades.COURSE_ID = Courses.ID)

UPDATE Courses SET AVG_GRADE = (SELECT AVG(GRADE) FROM Grades WHERE Grades.COURSE_ID = Courses.ID)

UPDATE Courses SET HIGHEST_GRADE = (SELECT MAX(GRADE) FROM Grades WHERE Grades.COURSE_ID = Courses.ID)
